#include <iostream.h>
#include <conio.h>

class InvalidMove
{
public:
    int pos;
    InvalidMove(int p)
    {
        pos = p;
    }
};

class Game
{
public:
    virtual void play() = 0;   // polymorphism
};

class TicTacToe : public Game   // inheritance
{
private:
    char board[9];
    int currentPlayer;

public:
    TicTacToe()
    {
        int i;
        for(i = 0; i < 9; i++)
            board[i] = '1' + i;

        currentPlayer = 1;
    }

    void showBoard()
    {
        cout << "\n\n";
        cout << " " << board[0] << " | " << board[1] << " | " << board[2] << "\n";
        cout << "---|---|---\n";
        cout << " " << board[3] << " | " << board[4] << " | " << board[5] << "\n";
        cout << "---|---|---\n";
        cout << " " << board[6] << " | " << board[7] << " | " << board[8] << "\n\n";
    }

    void makeMove(int pos)
    {
        if(pos < 1 || pos > 9)
            throw InvalidMove(pos);

        if(board[pos - 1] == 'X' || board[pos - 1] == 'O')
            throw InvalidMove(pos);

        if(currentPlayer == 1)
            board[pos - 1] = 'X';
        else
            board[pos - 1] = 'O';
    }

    int checkWin()
    {
        int i;
        int winLines[8][3] = {
            {0,1,2}, {3,4,5}, {6,7,8},
            {0,3,6}, {1,4,7}, {2,5,8},
            {0,4,8}, {2,4,6}
        };

        for(i = 0; i < 8; i++)
        {
            int a = winLines[i][0];
            int b = winLines[i][1];
            int c = winLines[i][2];

            if(board[a] == board[b] && board[b] == board[c])
            {
                if(board[a] == 'X')
                    return 1;
                if(board[a] == 'O')
                    return 2;
            }
        }

        for(i = 0; i < 9; i++)
        {
            if(board[i] != 'X' && board[i] != 'O')
                return 0;   // game not finished
        }

        return 3; // draw
    }

    void play()   // polymorphism
    {
        int pos, result = 0;

        while(1)
        {
            clrscr();
            cout << "TIC TAC TOE GAME\n";
            cout << "Player 1 = X, Player 2 = O\n";
            showBoard();

            cout << "Player " << currentPlayer << ", enter position (1-9): ";
            cin >> pos;

            try
            {
                makeMove(pos);
            }
            catch(InvalidMove e)
            {
                cout << "\nInvalid move! Try position " << e.pos << " again.";
                getch();
                continue;
            }

            result = checkWin();

            if(result == 1)
            {
                clrscr();
                showBoard();
                cout << "Player 1 (X) wins!";
                break;
            }
            else if(result == 2)
            {
                clrscr();
                showBoard();
                cout << "Player 2 (O) wins!";
                break;
            }
            else if(result == 3)
            {
                clrscr();
                showBoard();
                cout << "Match Draw!";
                break;
            }

            if(currentPlayer == 1)
                currentPlayer = 2;
            else
                currentPlayer = 1;
        }
    }
};

void main()
{
    Game *g;          // object pointer
    TicTacToe t;      // object

    clrscr();
    g = &t;          // polymorphism
    g->play();

    getch();
}
